  --conf spark.hadoop.fs.s3a.access.key=minioadmin \

-- Create Iceberg table from Parquet files
CREATE TABLE customers
USING iceberg
AS
SELECT 
  after.ID as id,
  after.NAME as name
FROM parquet.`s3a://oracle-cdc/topics/server1.C__DBZUSER.CUSTOMERS`;

-- Query the table
SELECT * FROM customers;
