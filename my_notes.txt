WITH latest_before_1_hour AS (
SELECT COALESCE(after.ID, before.ID) as id,COALESCE(after.NAME, before.NAME) as name,op,ts_ms,ROW_NUMBER() OVER (PARTITION BY COALESCE(after.ID, before.ID) 
ORDER BY ts_ms DESC as rn FROM parquet.`s3a://oracle-cdc/topics/server1.C__DBZUSER.CUSTOMERS` WHERE ts_ms < (UNIX_TIMESTAMP(CURRENT_TIMESTAMP() - INTERVAL '2' HOURS) * 1000))
SELECT id,name,'state as of 1 hour ago' as description FROM latest_before_1_hour WHERE rn = 1 AND op != 'd' ORDER BY id;

# Stop both connectors
curl -X PUT http://localhost:8083/connectors/customers-connector/stop
curl -X PUT http://localhost:8083/connectors/minio-s3-sink/stop
# Reset offsets for source connector (customers-connector)
curl -X DELETE http://localhost:8083/connectors/customers-connector/offsets
# For sink connector, you need to reset consumer group offsets
docker exec -it kafka kafka-consumer-groups \
  --bootstrap-server kafka:9092 \
  --group connect-minio-s3-sink \
  --reset-offsets --to-earliest --execute --all-topics
# Resume both connectors
curl -X PUT http://localhost:8083/connectors/customers-connector/resume
curl -X PUT http://localhost:8083/connectors/minio-s3-sink/resume
curl -X DELETE http://localhost:8083/connectors/customers-connector
curl -X DELETE http://localhost:8083/connectors/minio-s3-sink
